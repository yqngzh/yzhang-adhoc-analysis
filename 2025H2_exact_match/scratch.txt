SELECT request, queryTime 
FROM `etsy-searchinfra-gke-prod-2.thrift_mmx_listingsv2search_search.rpc_logs_*` logs 
WHERE request.OPTIONS.cacheBucketId = "live|web" 
AND queryTime >= "2025-08-22 03:00:00+00:00" 
AND queryTime < "2025-08-22 04:00:00+00:00" 
AND request.options.queryType NOT IN ("prolist", "prolist_similar", "prolist_user_to_listing") 
AND request.options.csrOrganic=true
AND request.offset = 0
AND (request.options.prolistOptions.useVespaANN = false OR request.options.prolistOptions.useVespaANN IS NULL) 
AND NOT EXISTS (SELECT * FROM UNNEST(request.context) WHERE key = "req_source" AND value = "bot") 
-- SO
AND request.options.searchPlacement in unnest(["wsg", "wmg"])
AND request.options.userCountry = "US"
AND request.options.personalizationOptions.userId = 0
-- SI
-- AND request.options.searchPlacement in unnest(["allsr", "wsg", "wmg"])
-- AND request.options.userCountry = "US"
-- AND request.options.personalizationOptions.userId != 0
AND request.options.solrRetrievalOnly = FALSE
AND OrganicRequestMetadata.candidateSources IS NOT NULL
AND runtime.sampleVal < 0.01
ORDER BY runtime.sampleVal 
LIMIT 13750
